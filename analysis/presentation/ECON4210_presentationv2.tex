% --- Setup ---
\begin{frame}{Setup: Tasks, Workers, and Production}

\begin{itemize}
  \item A continuum of cognitive \textbf{tasks} $i\in[0,1]$, ordered by
        susceptibility to AI

  \medskip
  \item \textbf{Workers} of type $i^*$ specialize in task $i^*$;
        human supply of task $i$ is $f(i)$ --- fixed and inelastic

  \medskip
  \item \textbf{CES production} aggregates all tasks into output:
  \[
    Y = \left[\int_0^1 y(i)^{\frac{\sigma-1}{\sigma}}\,di
        \right]^{\!\frac{\sigma}{\sigma-1}},
    \qquad \sigma > 0
  \]

  \medskip
  \item Output price $P_Y$ fixed (small open economy --- think: world price)
\end{itemize}

\end{frame}

% --- Pre-AI Wage ---
\begin{frame}{Pre-AI Wages}

Firms minimize cost $\rightarrow$ market clearing $\rightarrow$ each worker
earns the competitive price of their task:
\[
  w_0(i^*) \;=\; P_Y\!\left(\frac{Y_0}{f(i^*)}\right)^{\!1/\sigma}
\]

\centering
\includegraphics[width=0.70\linewidth]{fig_preai.png}

\smallskip
{\small Scarce tasks pay more; abundant tasks pay less. }

{\small \textcolor{red}{Note: Task supply $f(i)$ decreasing in $i$ is not necessary}}

\end{frame}

% --- How AI enters ---
\begin{frame}{How AI Enters: A Supply Shock}

\textbf{AI adds supply} $s(i)\geq 0$ to the tasks it can perform:
\[
  s'(i) < 0 \text{ for } i<I^*,
  \qquad
  s(i) = 0 \text{ for } i\geq I^*
\]

\begin{columns}[c]
\column{0.55\textwidth}
  \begin{itemize}
    \item $I^*$ = \textbf{AI capability frontier}
    \item AI is best at routine tasks (low $i$),
          contribution shrinks as $i\to I^*$
    \item Tasks above $I^*$: beyond AI's reach
    \item As AI improves, $I^*$ shifts right
  \end{itemize}

\column{0.42\textwidth}
  \centering
  \includegraphics[width=0.90\linewidth]{toy_model.png}\\
  {\small\color{mygray} Think: offshoring your tasks\\to Claudeland}
\end{columns}

\end{frame}

% =============================================================================

% --- Post-AI Wage Equation ---
\begin{frame}{Post-AI Wages}

Effective supply: $\bar y(i) = f(i) + s(i)$.  Output expands to
$\bar Y > Y_0$.

\medskip
Same CES logic gives the \textbf{post-AI wage}:
\[
  \boxed{w_1(i^*) \;=\; P_Y\!\left(\frac{\bar Y}{f(i^*)+s(i^*)}
        \right)^{\!1/\sigma}}
\]

\medskip
Compare to pre-AI: $w_0 = P_Y(Y_0/f)^{1/\sigma}$.  Two things moved:
\begin{itemize}
  \item[\textcolor{mygreen}{$\uparrow$}] \textbf{Numerator} $\bar Y > Y_0$:
        economy grew $\rightarrow$ helps \emph{everyone}
  \item[\textcolor{myred}{$\downarrow$}] \textbf{Denominator} $f{+}s > f$ for
        $i<I^*$: more supply $\rightarrow$ hurts exposed workers
\end{itemize}

\medskip
Who wins and who loses depends on which effect dominates.

\end{frame}

% --- Three Groups ---
\begin{frame}{Three Groups of Workers}

\begin{columns}[t]
\column{0.32\textwidth}
  \textbf{\textcolor{mygreen}{Above $I^*$}}\\[4pt]
  $s=0$, denominator unchanged.
  \[
    \frac{w_1}{w_0} = \left(\frac{\bar Y}{Y_0}\right)^{\!1/\sigma}\!> 1
  \]
  Pure gain from output expansion.

\column{0.34\textwidth}
  \textbf{\textcolor{myred}{Deep inside $[0,I^*)$}}\\[4pt]
  $s/f$ large --- denominator grows faster than numerator.
  \[
    \frac{s(i^*)}{f(i^*)} > \frac{\bar Y}{Y_0} - 1
  \]
  Wage compression.  If below reservation $\rightarrow$ displacement.

\column{0.32\textwidth}
  \textbf{\textcolor{mygreen}{Near $I^*$ from below}}\\[4pt]
  $s(i^*)\approx 0$, so denominator barely moved.
  \[
    \frac{w_1}{w_0} \approx \left(\frac{\bar Y}{Y_0}\right)^{\!1/\sigma}\!> 1
  \]
  Still gain --- split is at $\hat\imath < I^*$, not at $I^*$.
\end{columns}

\end{frame}

% --- B1: Pre-AI Derivation ---
\begin{frame}{Derivation: Pre-AI Equilibrium}

\textbf{Step 1.} Cost minimization $\rightarrow$ FOC:
$\;p_0(i)=\mu\cdot(Y/y(i))^{1/\sigma}$, \quad $\mu=P_Y$ in competitive eq.

\medskip
\textbf{Step 2.} Task demand: $\;y^d(i)=Y(P_Y/p_0(i))^{\sigma}$

\medskip
\textbf{Step 3.} Market clearing $y^d(i)=f(i)$:
\[
  p_0(i) = P_Y\!\left(\frac{Y}{f(i)}\right)^{\!1/\sigma}
\]

\textbf{Step 4.} Plug $y(i)=f(i)$ into CES $\rightarrow$ output is pinned:
\[
  \boxed{Y_0 = \left[\int_0^1 f(i)^{(\sigma-1)/\sigma}\,di
  \right]^{\!\sigma/(\sigma-1)}}
\]

$Y_0$ depends only on worker distribution $f$ --- not on prices.

\end{frame}

% --- B2: Post-AI Derivation ---
\begin{frame}{Derivation: Post-AI Equilibrium}

\textbf{Step 5.} AI adds supply $s(i)$. Effective supply:
$\bar y(i)=f(i)+s(i)$.

\medskip
Repeat Steps 1--4 with $\bar y$ in place of $f$:
\[
  \bar Y = \left[\int_0^1 (f(i)+s(i))^{(\sigma-1)/\sigma}\,di
  \right]^{\!\sigma/(\sigma-1)} > Y_0
\]

Post-AI wage:
\[
  \boxed{w_1(i^*) = P_Y\!\left(\frac{\bar Y}{f(i^*)+s(i^*)}\right)^{\!1/\sigma}}
\]

\medskip
Wage ratio (Group 2 formula):
\[
  \frac{w_1(i^*)}{w_0(i^*)}
  = \underbrace{\left(\frac{\bar Y}{Y_0}\right)^{\!1/\sigma}}_{\text{output expansion} > 1}
  \cdot
  \underbrace{\left(\frac{f(i^*)}{f(i^*)+s(i^*)}\right)^{\!1/\sigma}}_{\text{supply effect} \leq 1}
\]

\end{frame}

% --- B3: Three-Group Analysis ---
\begin{frame}{Three-Group Analysis (Formal)}

\small
\textbf{Group 1} ($i^*\geq I^*$, $s=0$): \quad
$w_1/w_0 = (\bar Y/Y_0)^{1/\sigma} > 1$ \quad
--- uniform gain, increasing in aggregate AI supply.

\bigskip
\textbf{Group 2} ($i^*<I^*$, $s/f$ large): \quad
Compression whenever $\;s(i^*)/f(i^*) > \bar Y/Y_0 - 1$.

Because $\bar Y/Y_0$ is a CES average over \emph{all} tasks (most with $s=0$),
while $s/f$ is concentrated on AI-supplied tasks, this condition holds for
heavily exposed workers.

\bigskip
\textbf{Group 3} ($i^*<I^*$, $s/f$ small): \quad
Near $I^*$, $s\to 0$ so supply effect $\approx 1$ and output expansion
dominates $\rightarrow$ these workers \textcolor{mygreen}{gain} despite
$i^*<I^*$.

\bigskip
\textbf{Threshold $\hat\imath < I^*$:}\quad
$s(\hat\imath)/f(\hat\imath) = \bar Y/Y_0 - 1$.\\
Winners: $i^*>\hat\imath$.  Losers: $i^*<\hat\imath$.
The boundary lies \emph{strictly below} $I^*$.

\end{frame}

% --- B4: Expanding Frontier + Non-Monotone f ---
\begin{frame}{Further Discussion of $I^*$}

\begin{columns}[t]
\column{0.50\textwidth}

  \textbf{As AI improves, $I^*$ shifts right}\\[4pt]
  More workers exposed, deeper compression, $\hat\imath$ shifts rightward.

  \medskip
  \includegraphics[width=\linewidth]{fig_capability.png}
  {\tiny Each curve: $w_1$ for a different $I^*$. Higher $I^*$ $\rightarrow$ wider compression.}

\column{0.48\textwidth}

  \textbf{Non-monotone $f(i)$: scattered effects}\\[4pt]
  If many workers cluster in a mid-complexity task, the local $s/f$ ratio
  drops, creating pockets where workers gain even deeper inside $[0,I^*)$.

  \medskip
  \includegraphics[width=\linewidth]{fig_multiple.png}

\end{columns}
