\begin{frame}{A Lesson from Trade Economics}

\textbf{Offshoring (2000s):} companies shipped tasks abroad when foreign labor was cheaper

\bigskip
\begin{itemize}
  \item \textbf{Grossman \& Rossi-Hansberg (2008):} it's not whole jobs that get offshored ---
  it's specific \emph{tasks}

  \bigskip
  \item \textbf{Ottaviano, Peri \& Wright (2013):} different workers face different cutoffs ---
  not everyone equally at risk

  \bigskip
  \item \textbf{AI is the same story, new technology:}\\[4pt]
    Instead of cheap foreign workers --- a cheap \emph{machine}.\\
    \textbf{``Offshoring'' your tasks to Claudeland, GPTopia, Geminia\ldots}
\end{itemize}

\begin{frame}{What the Model Predicts}

\textbf{Workers to the left of $I^*$ face two immediate pressures:}

\bigskip
\begin{columns}[t]
\column{0.48\textwidth}
  \textbf{\textcolor{myred}{Wage compression}}\\[6pt]
  Firms can switch to AI $\rightarrow$ workers must accept lower pay to stay employed

\column{0.48\textwidth}
  \textbf{\textcolor{myred}{Displacement}}\\[6pt]
  If the compressed wage drops below what you're willing to accept $\rightarrow$ you exit

\end{columns}

\bigskip\hrule\bigskip

{\small\color{mygray}
\textit{This captures the \textbf{immediate substitution effect} only ---
longer-run channels (augmentation, output expansion, new tasks) could go the other way.}}

\end{frame}

% --- B1: Model Setup ---
\begin{frame}{Model Setup (Formal)}

\begin{itemize}
  \item \textbf{Tasks} indexed $i \in [0,1]$ by AI susceptibility
  \begin{itemize}
    \item Low $i$: routine cognitive --- easy for AI \quad|\quad High $i$: complex --- hard for AI \textit{currently}
  \end{itemize}

  \medskip
  \item \textbf{CES Production}:
  \[
    Y = \left[\int_0^1 y(i)^{\frac{\sigma-1}{\sigma}}\,di\right]^{\!\frac{\sigma}{\sigma-1}}
  \]

  \medskip
  \item \textbf{Workers:} type $i^*$ specializes in task $i^*$; supply of task $i$ is $f(i)$ (fixed, inelastic)

  \medskip
  \item \textbf{AI cost} of task $i$: $\quad c^{\mathrm{AI}}(i) = \tau\cdot e^{\alpha i}$
  \begin{itemize}
    \item Lower $\tau$ = more capable AI; $e^{\alpha i}$ ensures complex tasks remain costly for AI
  \end{itemize}

  \medskip
  \item Competitive firms; output price $P_Y$ fixed (small open economy)
\end{itemize}

\end{frame}

% --- B2: Pre-AI Derivation ---
\begin{frame}{Pre-AI Equilibrium: Derivation}

\textbf{Firm cost minimization gives task demand:}
\[
  y^d(i) \;=\; Y \cdot \left(\frac{P_Y}{p_0(i)}\right)^{\!\sigma}
\]

\textbf{Market clearing} $\bigl(y^d(i) = f(i)\bigr)$ pins down task price:
\[
  p_0(i) \;=\; P_Y \cdot \left(\frac{Y}{f(i)}\right)^{\!1/\sigma}
\]

\textbf{Solve for $Y_0$} --- plug $y(i)=f(i)$ back into the CES aggregator:
\[
  \boxed{Y_0 \;=\; \left[\int_0^1 f(i)^{\frac{\sigma-1}{\sigma}}\,di\right]^{\!\frac{\sigma}{\sigma-1}}}
\]

\begin{itemize}
  \item $Y_0$ is fully determined by worker distribution $f$ --- not by prices
  \item $Y_0$ is the output level held fixed in the short-run AI shock analysis
\end{itemize}

\end{frame}

% --- B3: Pre-AI Wage Schedule ---
\begin{frame}{Pre-AI Equilibrium: Wage Schedule}

Under full specialization, each worker earns their task price:
\[
  w_0(i^*) \;=\; p_0(i^*) \;=\; P_Y \cdot \left(\frac{Y_0}{f(i^*)}\right)^{\!1/\sigma}
\]

\begin{columns}[c]
\column{0.58\textwidth}
  \includegraphics[width=\linewidth]{fig_preai.png}
\column{0.38\textwidth}
  \small
  \begin{itemize}
    \item Each worker earns the competitive price of their task
    \item Scarce types earn more; abundant types earn less
    \item[\phantom{x}]~
    \item[] \textit{Upward slope for illustration only --- no shape restriction on $p_0$}
  \end{itemize}
\end{columns}

\end{frame}

% --- B4: Immediate Effects Framing ---
\begin{frame}{Immediate Effects: What We Do and Don't Model}

\textcolor{myred}{\textbf{Hold $Y = Y_0$ fixed --- immediate impact before the economy adjusts}}

\bigskip
Isolates the \textbf{substitution channel} only. Longer-run channels set aside:
\begin{itemize}
  \item \textbf{Augmentation} --- AI complements some workers, raising productivity over time
  \item \textbf{Output expansion} --- cheaper production eventually raises $Y$ beyond $Y_0$
  \item \textbf{Task creation} --- AI creates new tasks over time (Acemoglu \& Restrepo 2019)
\end{itemize}

\bigskip
Immediate task price once AI enters:
\[
  p(i) = \min\!\bigl(p_0(i),\;\tau e^{\alpha i}\bigr)
  \qquad \Rightarrow \qquad
  \text{cutoff } I^*:\; \tau e^{\alpha I^*} = p_0(I^*)
\]

\end{frame}

% --- B5: Cutoff and Wages ---
\begin{frame}{Immediate Effects: Cutoff and Wages}

\begin{columns}[c]
\column{0.55\textwidth}
  \includegraphics[width=\linewidth]{fig_postai.png}
\column{0.42\textwidth}
  \small
  \begin{itemize}
    \item $i < I^*$: AI immediately undercuts\\
          $\rightarrow$ \textcolor{myred}{wage compression}
    \[
      \Delta w = \tau e^{\alpha i^*} - p_0(i^*) < 0
    \]
    \item[\phantom{x}]~
    \item $i \geq I^*$: no direct price pressure \textcolor{mygreen}{in the short run}
    \begin{itemize}
      \item[\scriptsize$\hookrightarrow$] {\small output expansion raises their wages over time}
    \end{itemize}
    \item[\phantom{x}]~
    \item Compressed wage $<$ reservation\\
          $\rightarrow$ \textbf{exit employment}
  \end{itemize}
\end{columns}

\end{frame}

% --- B6: More on I* ---
\begin{frame}{Further Discussion of $I^*$}

\begin{columns}[t]
\column{0.50\textwidth}

  \textbf{As AI improves ($\tau \downarrow$), $I^*$ rises}
  \[
    \frac{\partial I^*}{\partial \tau} < 0
  \]
  The automation frontier expands toward more complex tasks.

  \medskip
  \includegraphics[width=\linewidth]{fig_capability.png}
  {\tiny Decreasing $\tau$ shifts the AI cost curve down, pushing $I^*$ rightward.}

\column{0.48\textwidth}

  \textbf{Non-monotone $p_0(i)$: multiple cutoffs}
  \[
    \mathcal{A} = [0, I_1^*] \cup [I_2^*, I_3^*] \cup \cdots
  \]
  Wage compression applies to all $i^* \in \mathcal{A}$; exposed workers need not be at the low end.

  \medskip
  \includegraphics[width=\linewidth]{fig_multiple.png}
  {\tiny Multiple crossings when the pre-AI wage schedule is non-monotone.}

\end{columns}
