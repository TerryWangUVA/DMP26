\begin{frame}{Building on Trade Theory}

\begin{itemize}
  \item \textbf{Task-based production:} firms combine a \emph{continuum of tasks}, not just labor and capital
  \begin{itemize}
    \item Tasks differ in how easily they can be delegated to a cheaper alternative
  \end{itemize}

  \bigskip
  \item \textbf{Grossman \& Rossi-Hansberg (2008) --- ``Offshoring Tasks''}
  
  \bigskip
  \item \textbf{Ottaviano, Peri \& Wright (2013)--- ``Who are affected''}


  \bigskip
  \item \textbf{Why does this apply to AI?}
  \begin{itemize}
    \item AI = a technology that can perform certain tasks at low cost
    \item \textbf{``offshoring'' (outsourcing) tasks
    to Claudeland, GPTopia, Geminia...}
    \item Workers whose tasks AI can undercut face a similar wage pressure as offshored workers
  \end{itemize}
\end{itemize}

\end{frame}

% =============================================================================
% SLIDE 3 — Model Setup
% =============================================================================
\begin{frame}{Model Setup}

\begin{itemize}
  \item \textbf{Tasks} indexed $i \in [0,1]$ by AI susceptibility
  \begin{itemize}
    \item Low $i$: routine cognitive (drafting, coding, data entry) --- easy for AI
    \item High $i$: human expertise --- hard for AI \textit{currently}
  \end{itemize}

  \medskip
  \item \textbf{CES Production}:
  \[
    Y = \left[\int_0^1 y(i)^{\frac{\sigma-1}{\sigma}}\,di\right]^{\!\frac{\sigma}{\sigma-1}}
  \]

  \medskip
  \item \textbf{Workers:} type $i^*$ specializes in task $i^*$; supply of task $i$ is $f(i)$ (fixed)

  \medskip
  \item \textbf{AI cost} of task $i$: $\quad c^{\mathrm{AI}}(i) = \tau\cdot e^{\alpha i}$
  \begin{itemize}
    \item Lower $\tau$ = more capable AI
    \item $e^{\alpha i}$ ensures complex tasks remain expensive for AI
  \end{itemize}

  \medskip
  \item Competitive firms; output price $P_Y$ fixed (small open economy)
\end{itemize}

\end{frame}

% =============================================================================
% SLIDE 4 — Pre-AI Equilibrium: Derivation
% =============================================================================
\begin{frame}{Pre-AI Equilibrium: Derivation}

\textbf{Firm cost minimization gives task demand:}
\[
  y^d(i) \;=\; Y \cdot \left(\frac{P_Y}{p_0(i)}\right)^{\!\sigma}
\]

\textbf{Market clearing} $\bigl(y^d(i) = f(i)\bigr)$ pins down task price:
\[
  p_0(i) \;=\; P_Y \cdot \left(\frac{Y}{f(i)}\right)^{\!1/\sigma}
\]

\textbf{Solve for $Y = Y_0$} by plugging $y(i)=f(i)$ back into the CES aggregator:
\[
  \boxed{Y_0 \;=\; \left[\int_0^1 f(i)^{\frac{\sigma-1}{\sigma}}\,di\right]^{\!\frac{\sigma}{\sigma-1}}}
\]

\begin{itemize}
  \item $Y_0$ is fully determined by the worker type distribution $f$ --- not by prices
  \item Scarce worker types (low $f(i)$) command higher task prices; abundant types earn less
  \item $Y_0$ is the output level we hold fixed in the short-run AI shock analysis
\end{itemize}

\end{frame}

% SLIDE 6a — Immediate Effects: Framing
% =============================================================================
\begin{frame}{Immediate Effects of the AI Shock}

\textcolor{myred}{\textbf{Hold $Y = Y_0$ fixed --- immediate impact before the economy adjusts}}

\bigskip
This isolates the \textbf{substitution channel} only.\\[4pt]
Longer-run channels deliberately set aside:

\bigskip
\begin{itemize}
  \item \textbf{Augmentation} --- AI complements some workers, raising their productivity over time
  \item \textbf{Output expansion} --- cheaper production eventually raises $Y$ beyond $Y_0$, benefiting non-automated workers
  \item \textbf{Task creation} --- AI creates new tasks over time (Acemoglu \& Restrepo 2019)
\end{itemize}

\bigskip
Immediate task price once AI enters:
\[
  p(i) = \min\!\bigl(p_0(i),\;\tau e^{\alpha i}\bigr)
  \qquad \Rightarrow \qquad
  \text{cutoff } I^*:\; \tau e^{\alpha I^*} = p_0(I^*)
\]

\end{frame}

% SLIDE 7 — More on I*

\begin{frame}{What Determines the Cutoff $I^*$?}

\begin{columns}[t]
\column{0.50\textwidth}

  \textbf{As AI improves ($\tau \downarrow$), $I^*$ rises}
  \[
    \frac{\partial I^*}{\partial \tau} < 0
  \]
  The automation frontier expands.

  \medskip
  \includegraphics[width=\linewidth]{fig_capability.png}
  {\tiny Decreasing $\tau$ shifts the AI cost curve down, pushing $I^*$ rightward.}

\column{0.48\textwidth}

  \textbf{Non-monotone $p_0(i)$}\\[2pt]
  Cutoff may not be unique $\rightarrow$ automated set is a union of intervals:
  \[
    \mathcal{A} = [0, I_1^*] \cup [I_2^*, I_3^*] \cup \cdots
  \]

  \medskip
  \includegraphics[width=\linewidth]{fig_multiple.png}
  {\tiny Wage compression applies to all $i^* \in \mathcal{A}$; exposed workers need not be at the low end.}

\end{columns}

\end{frame}
