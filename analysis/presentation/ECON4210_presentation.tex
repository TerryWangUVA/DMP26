% SLIDE 2 — Inspiration

\begin{frame}{Building on Trade Theory}

\begin{itemize}
  \item \textbf{Task-based production:} firms combine a \emph{continuum of tasks}, not just labor and capital
  \begin{itemize}
    \item Tasks differ in how easily they can be delegated to a cheaper alternative
  \end{itemize}

  \bigskip
  \item \textbf{Grossman \& Rossi-Hansberg (2008) --- ``Trading Tasks''}
  \begin{itemize}
    \item Offshoring = delegating tasks to cheaper foreign labor
    \item A falling offshoring cost $\rightarrow$ more tasks sent abroad $\rightarrow$ wage pressure on affected workers
  \end{itemize}

  \bigskip
  \item \textbf{Ottaviano, Peri \& Wright (2013)}
  \begin{itemize}
    \item Workers differ in type $i^*$; different groups face different offshoring cutoffs
    \item Heterogeneous exposure: not everyone is equally at risk
  \end{itemize}

  \bigskip
  \item \textbf{Why does this apply to AI?}
  \begin{itemize}
    \item AI = a technology that can perform certain tasks at low cost
    \item Replace ``offshoring cost'' with ``AI cost'' --- the mechanism is identical
    \item Workers whose tasks AI can undercut face the same wage pressure as offshored workers
  \end{itemize}
\end{itemize}

\end{frame}

% =============================================================================
% SLIDE 3 — Model Setup
% =============================================================================
\begin{frame}{Model Setup}

\begin{itemize}
  \item \textbf{Tasks} indexed $i \in [0,1]$ by AI susceptibility
  \begin{itemize}
    \item Low $i$: routine cognitive (drafting, coding, data entry) --- easy for AI
    \item High $i$: complex, judgment-intensive --- hard for AI
  \end{itemize}

  \medskip
  \item \textbf{Production} via CES aggregator:
  \[
    Y = \left[\int_0^1 y(i)^{\frac{\sigma-1}{\sigma}}\,di\right]^{\!\frac{\sigma}{\sigma-1}}
  \]

  \medskip
  \item \textbf{Workers:} type $i^*$ specializes in task $i^*$; supply of task $i$ is $f(i)$ (fixed)

  \medskip
  \item \textbf{AI cost} of task $i$: $\quad c^{\mathrm{AI}}(i) = \tau\cdot e^{\alpha i}$
  \begin{itemize}
    \item Lower $\tau$ = more capable AI
    \item $e^{\alpha i}$ ensures complex tasks remain expensive for AI
  \end{itemize}

  \medskip
  \item Competitive firms; output price $P_Y$ fixed (small open economy)
\end{itemize}

\end{frame}

% =============================================================================
% SLIDE 4 — Pre-AI Equilibrium: Derivation
% =============================================================================
\begin{frame}{Pre-AI Equilibrium: Derivation}

\textbf{Step 1 --- Firm cost minimization gives task demand:}
\[
  y^d(i) \;=\; Y \cdot \left(\frac{P_Y}{p_0(i)}\right)^{\!\sigma}
\]
Higher task price $\rightarrow$ firm demands less of that task (standard substitution).

\bigskip
\textbf{Step 2 --- Market clearing} $\bigl(y^d(i) = f(i)\bigr)$ pins down equilibrium task price:
\[
  \boxed{p_0(i) \;=\; P_Y \cdot \left(\frac{Y}{f(i)}\right)^{\!1/\sigma}}
\]

\begin{itemize}
  \item Scarce worker types (low $f(i)$) command higher task prices
  \item Abundant worker types (high $f(i)$) earn less
\end{itemize}

\end{frame}

\begin{frame}{Pre-AI Equilibrium: Wage Schedule}

\textbf{Step 3 --- Under full specialization, each worker earns their task price:}
\[
  w_0(i^*) \;=\; p_0(i^*)
\]

% =============================================================================
% SLIDE 6 — Post-AI Shock
% =============================================================================
\begin{frame}{Post-AI Shock: The Automation Cutoff}

\textbf{AI enters as a competing supplier.} Competition drives task prices to:
\[
  p(i) = \min\!\bigl(p_0(i),\;\tau e^{\alpha i}\bigr)
\]

\textbf{A cutoff $I^*$ emerges} where AI cost equals pre-AI task price: $\;\tau e^{\alpha I^*} = p_0(I^*)$

\begin{columns}[c]
\column{0.55\textwidth}
  \includegraphics[width=\linewidth]{fig_postai.png}
\column{0.42\textwidth}
  \small
  \begin{itemize}
    \item $i < I^*$: AI undercuts $\rightarrow$ \textcolor{myred}{wage compression}
    \[
      \Delta w(i^*) = \tau e^{\alpha i^*} - p_0(i^*) < 0
    \]
    \item $i \geq I^*$: human price prevails $\rightarrow$ \textcolor{mygreen}{unaffected}
    \item[\phantom{x}]~
    \item Compressed wage $<$ reservation wage $\rightarrow$ \textbf{exit employment}
  \end{itemize}
\end{columns}

\end{frame}

% =============================================================================
% SLIDE 7 — More on I*
% =============================================================================
\begin{frame}{What Determines the Cutoff $I^*$?}

\begin{columns}[t]
\column{0.50\textwidth}

  \textbf{As AI improves ($\tau \downarrow$), $I^*$ rises}
  \[
    \frac{\partial I^*}{\partial \tau} < 0
  \]
  The automation frontier expands toward more complex tasks.

  \medskip
  \includegraphics[width=\linewidth]{fig_capability.png}
  {\tiny Decreasing $\tau$ shifts the AI cost curve down, pushing $I^*$ rightward.}

\column{0.48\textwidth}

  \textbf{What if $p_0(i)$ is non-monotone?}\\[2pt]
  Cutoff may not be unique $\rightarrow$ automated set is a union of intervals:
  \[
    \mathcal{A} = [0, I_1^*] \cup [I_2^*, I_3^*] \cup \cdots
  \]

  \medskip
  \includegraphics[width=\linewidth]{fig_multiple.png}
  {\tiny Wage compression applies to all $i^* \in \mathcal{A}$; exposed workers need not be at the low end.}

\end{columns}

\end{frame}
